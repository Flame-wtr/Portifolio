                    int LDR = A0;

                      pinMode(LED, OUTPUT);
                      pinMode(LDR, INPUT);
                      Serial.begin(9600);
                    }
                    //Repetições
                    void loop() {
                      int luz = analogRead(LDR);
                      Serial.println(luz);

                        digitalWrite(LED, HIGH);
                      } 
                      else {
                        digitalWrite(LED, LOW);
